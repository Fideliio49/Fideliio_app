Fix 1 — Enable Row Level Security (RLS) on all tables in Supabase :
The Supabase Security Advisor is showing 5 critical errors :

public.merchants — RLS Disabled
public.customers — RLS Disabled
public.rewards — RLS Disabled
public.transactions — RLS Disabled
public.redemptions — RLS Disabled

Run these SQL commands in the Supabase SQL Editor :
sql-- Enable RLS on all tables
ALTER TABLE public.merchants ENABLE ROW LEVEL SECURITY;
ALTER TABLE public.customers ENABLE ROW LEVEL SECURITY;
ALTER TABLE public.rewards ENABLE ROW LEVEL SECURITY;
ALTER TABLE public.transactions ENABLE ROW LEVEL SECURITY;
ALTER TABLE public.redemptions ENABLE ROW LEVEL SECURITY;

-- Merchants : a merchant can only read/write their own data
CREATE POLICY "Merchant access own data"
ON public.merchants
FOR ALL
USING (auth.uid() = user_id);

-- Customers : a customer can only read/write their own data
CREATE POLICY "Customer access own data"
ON public.customers
FOR ALL
USING (auth.uid() = user_id);

-- Transactions : customer can read their own, merchant can read/write their own
CREATE POLICY "Customer read own transactions"
ON public.transactions
FOR SELECT
USING (
  customer_id IN (
    SELECT id FROM public.customers WHERE user_id = auth.uid()
  )
);

CREATE POLICY "Merchant write own transactions"
ON public.transactions
FOR INSERT
WITH CHECK (
  merchant_id IN (
    SELECT id FROM public.merchants WHERE user_id = auth.uid()
  )
);

CREATE POLICY "Merchant read own transactions"
ON public.transactions
FOR SELECT
USING (
  merchant_id IN (
    SELECT id FROM public.merchants WHERE user_id = auth.uid()
  )
);

-- Rewards : merchant manages their own, customers can read all active rewards
CREATE POLICY "Merchant manage own rewards"
ON public.rewards
FOR ALL
USING (
  merchant_id IN (
    SELECT id FROM public.merchants WHERE user_id = auth.uid()
  )
);

CREATE POLICY "Customer read active rewards"
ON public.rewards
FOR SELECT
USING (is_active = true);

-- Redemptions : customer and merchant can read their own
CREATE POLICY "Customer read own redemptions"
ON public.redemptions
FOR ALL
USING (
  customer_id IN (
    SELECT id FROM public.customers WHERE user_id = auth.uid()
  )
);

-- Merchants can read all customers (needed for QR scan)
CREATE POLICY "Merchant read customers for scan"
ON public.customers
FOR SELECT
USING (
  EXISTS (
    SELECT 1 FROM public.merchants WHERE user_id = auth.uid()
  )
);

-- All authenticated users can read merchants list
CREATE POLICY "Anyone can read merchants"
ON public.merchants
FOR SELECT
USING (auth.role() = 'authenticated');
