  try {
    setLoading(true);
    const { error } = await supabase.auth.signInWithOtp({
      email: email.trim(),
      options: { shouldCreateUser: false }
      // shouldCreateUser: false = login only, no new account
    });
    if (error) throw error;

    navigation.navigate('OTPVerification', {
      contact: email.trim(),
      contactType: 'email',
      userType: userType, // 'customer' or 'merchant'
      isLogin: true // flag to skip profile creation
    });
  } catch (error: any) {
    const msg = error?.message || '';
    if (msg.includes('not found') || msg.includes('Invalid')) {
      Alert.alert('Erreur', 'Aucun compte trouvé avec cet email.');
    } else {
      Alert.alert('Erreur', msg);
    }
  } finally {
    setLoading(false);
  }
};

typescript// The screen receives isLogin prop
// If isLogin = true → just verify OTP, do NOT create profile
// If isLogin = false → verify OTP + create profile

const onVerify = async () => {
  try {
    setLoading(true);

    let data;
    if (contactType === 'email') {
      const result = await supabase.auth.verifyOtp({
        email: contact,
        token: otp,
        type: 'email'
      });
      if (result.error) throw result.error;
      data = result.data;
    } else {
      const result = await supabase.auth.verifyOtp({
        phone: contact,
        token: otp,
        type: 'sms'
      });
      if (result.error) throw result.error;
      data = result.data;
    }

    // Only create profile if registering (not logging in)
    if (!isLogin) {
      await createProfileIfNotExists(
        data.user!.id,
        userData,
        userType
      );
    }

    // Navigate to correct dashboard
    if (userType === 'customer') {
      navigation.reset({
        index: 0,
        routes: [{ name: 'CustomerHome' }]
      });
    } else {
      navigation.reset({
        index: 0,
        routes: [{ name: 'MerchantDashboard' }]
      });
    }

  } catch (error: any) {
    const msg = error?.message || '';
    if (msg.includes('expired')) {
      setError('Code expiré. Demandez un nouveau code.');
    } else if (msg.includes('invalid')) {
      setError('Code incorrect. Vérifiez et réessayez.');
    } else {
      setError(msg || 'Erreur de vérification.');
    }
  } finally {
    setLoading(false);
  }
};
