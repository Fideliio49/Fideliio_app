Step 1 — Install the required package :
bashnpx expo install react-native-keyboard-aware-scroll-view

Step 2 — Android config :
In android/app/src/main/AndroidManifest.xml, set :
xml<activity
  android:windowSoftInputMode="adjustResize"
  ...
/>
In app.json :
json{
  "expo": {
    "android": {
      "softwareKeyboardLayoutMode": "pan"
    }
  }
}

Step 3 — iOS keyboard toolbar (add ONCE in App.tsx) :
javascriptimport {
  InputAccessoryView,
  Keyboard,
  Button,
  View,
  Platform
} from 'react-native';

export const KEYBOARD_TOOLBAR_ID = 'globalKeyboardDismiss';

// Add this inside the root App component return, after NavigationContainer :
{Platform.OS === 'ios' && (
  <InputAccessoryView nativeID={KEYBOARD_TOOLBAR_ID}>
    <View style={{
      backgroundColor: '#F8F8F8',
      borderTopWidth: 0.5,
      borderTopColor: '#E0E0E0',
      paddingHorizontal: 16,
      paddingVertical: 6,
      alignItems: 'flex-end'
    }}>
      <Button
        title="Fermer"
        onPress={Keyboard.dismiss}
      />
    </View>
  </InputAccessoryView>
)}

In each of these files, apply this pattern :
javascript// REMOVE :
import { ScrollView } from 'react-native';

// ADD :
import { KeyboardAwareScrollView } from 'react-native-keyboard-aware-scroll-view';

// REPLACE every <ScrollView> with :
<KeyboardAwareScrollView
  enableOnAndroid={true}
  enableAutomaticScroll={true}
  extraScrollHeight={Platform.OS === 'ios' ? 20 : 60}
  keyboardShouldPersistTaps="handled"
  showsVerticalScrollIndicator={false}
  contentContainerStyle={{ paddingBottom: 100 }}
>
  {/* existing content unchanged */}
</KeyboardAwareScrollView>
